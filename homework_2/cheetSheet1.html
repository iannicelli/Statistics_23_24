                // Creazione di una lista ordinata (sorted list)
                const sortedList = [5, 9, 2, 1, 8];

                // Ordina la lista in ordine crescente
                sortedList.sort((a, b) => a - b);

                // Stampa gli elementi della lista con un ciclo for
                console.log("Elementi nella lista ordinata:");
                for (let i = 0; i < sortedList.length; i++) {
                    console.log(sortedList[i]);
                }

                // Aggiungi un elemento alla lista mantenendo l'ordinamento con un ciclo for
                function insertSorted(arr, element) {
                    for (let i = 0; i < arr.length; i++) {
                        if (element < arr[i]) {
                            arr.splice(i, 0, element);
                            return; // L'elemento è stato inserito, esci dalla funzione
                        }
                    }

                // Se si arriva qui, l'elemento è il più grande, quindi aggiungilo alla fine
                arr.push(element);
                }

                insertSorted(sortedList, 4); // Aggiunge 4 alla lista in modo ordinato

                // Rimuovi un elemento dalla lista con un ciclo for
                const elementToRemove = 5;
                for (let i = 0; i < sortedList.length; i++) {
                    if (sortedList[i] === elementToRemove) {
                        sortedList.splice(i, 1);
                        break; // L'elemento è stato rimosso, esci dal ciclo
                    }
                }

                // Cerca un elemento nella lista con un ciclo for
                const elementToFind = 8;
                let isElementFound = false;
                for (let i = 0; i < sortedList.length; i++) {
                    if (sortedList[i] === elementToFind) {
                        isElementFound = true;
                        break; // Elemento trovato, esci dal ciclo
                    }
                }

                console.log("Lista ordinata:", sortedList);
                console.log("Elemento 4 aggiunto:", sortedList);
                console.log("Elemento 5 rimosso:", sortedList);
                console.log("Elemento 8 trovato:", isElementFound);

                class HashSet {
                    constructor() {
                      this.items = {};
                    }
                  
                    add(item) {
                      this.items[item] = true;
                    }
                  
                    remove(item) {
                      delete this.items[item];
                    }
                  
                    contains(item) {
                      return this.items[item] !== undefined;
                    }
                  
                    toArray() {
                      return Object.keys(this.items);
                    }
                  
                    // Restituisce un nuovo HashSet contenente l'intersezione tra due HashSet
                    intersection(otherSet) {
                      const intersectionSet = new HashSet();
                      const values = this.toArray();
                      
                      for (let i = 0; i < values.length; i++) {
                        if (otherSet.contains(values[i])) {
                          intersectionSet.add(values[i]);
                        }
                      }
                      
                      return intersectionSet;
                    }
                  
                    // Restituisce un nuovo HashSet contenente l'unione di due HashSet
                    union(otherSet) {
                      const unionSet = new HashSet();
                      const values = this.toArray();
                      
                      for (let i = 0; i < values.length; i++) {
                        unionSet.add(values[i]);
                      }
                  
                      values = otherSet.toArray();
                      
                      for (let i = 0; i < values.length; i++) {
                        unionSet.add(values[i]);
                      }
                  
                      return unionSet;
                    }
                }
                // Esempio di utilizzo della classe HashSet
                const myHashSet = new HashSet();

                myHashSet.add('element1');
                myHashSet.add('element2');
                myHashSet.add('element3');

                console.log("Contiene 'element2'?", myHashSet.contains('element2'));

                myHashSet.remove('element1');

                console.log("Contiene 'element1' dopo la rimozione?", myHashSet.contains('element1'));

                console.log("Elementi nell'HashSet:", myHashSet.toArray());

                class SortedSet {
                    constructor() {
                      this.items = [];
                    }
                  
                    insert(element) {
                      if (!this.contains(element)) {
                        this.items.push(element);
                        this.items.sort((a, b) => a - b);
                      }
                    }
                  
                    remove(element) {
                      const index = this.items.indexOf(element);
                      if (index !== -1) {
                        this.items.splice(index, 1);
                      }
                    }
                  
                    contains(element) {
                      return this.items.includes(element);
                    }
                  
                    size() {
                      return this.items.length;
                    }
                  
                    toArray() {
                      return this.items.slice();
                    }
                  }
                  
                  // Esempio di utilizzo della classe SortedSet
                  const mySortedSet = new SortedSet();
                  
                  mySortedSet.insert(3);
                  mySortedSet.insert(1);
                  mySortedSet.insert(4);
                  mySortedSet.insert(2);
                  mySortedSet.insert(5);
                  
                  console.log("Elementi nell'insieme ordinato:", mySortedSet.toArray()); // Output: [1, 2, 3, 4, 5]
                  
                  mySortedSet.remove(3);
                  console.log("Dopo la rimozione di 3:", mySortedSet.toArray()); // Output: [1, 2, 4, 5]
                  
                  const isPresent = mySortedSet.contains(4);
                  console.log(`È presente 4: ${isPresent}`); // Output: "È presente 4: true"
                  
                  console.log(`Dimensione dell'insieme: ${mySortedSet.size()}`); // Output: "Dimensione dell'insieme: 4"

                class Queue {
                    constructor() {
                      this.items = [];
                    }
                  
                    enqueue(element) {
                      this.items.push(element);
                    }
                  
                    dequeue() {
                      if (this.isEmpty()) {
                        return "La coda è vuota.";
                      }
                      return this.items.shift();
                    }
                  
                    front() {
                      if (this.isEmpty()) {
                        return "La coda è vuota.";
                      }
                      return this.items[0];
                    }
                  
                    isEmpty() {
                      return this.items.length === 0;
                    }
                  
                    size() {
                      return this.items.length;
                    }
                  }
                  
                  // Esempio di utilizzo della classe Queue
                  const myQueue = new Queue();
                  
                  myQueue.enqueue(1);
                  myQueue.enqueue(2);
                  myQueue.enqueue(3);
                  
                  console.log("Elemento in testa alla coda:", myQueue.front()); // Output: 1
                  
                  myQueue.dequeue();
                  console.log("Dopo dequeue, elemento in testa alla coda:", myQueue.front()); // Output: 2
                  
                  console.log(`La coda è vuota? ${myQueue.isEmpty()}`); // Output: "La coda è vuota? false"
                  
                  console.log(`Dimensione della coda: ${myQueue.size()}`); // Output: "Dimensione della coda: 2"

                class Stack {
                    constructor() {
                      this.items = [];
                    }
                  
                    push(element) {
                      this.items.push(element);
                    }
                  
                    pop() {
                      if (this.isEmpty()) {
                        return "Lo stack è vuoto.";
                      }
                      return this.items.pop();
                    }
                  
                    peek() {
                      if (this.isEmpty()) {
                        return "Lo stack è vuoto.";
                      }
                      return this.items[this.items.length - 1];
                    }
                  
                    isEmpty() {
                      return this.items.length === 0;
                    }
                  
                    clear() {
                      this.items = [];
                    }
                  
                    search(element) {
                      const index = this.items.indexOf(element);
                      return index === -1 ? -1 : this.items.length - index;
                    }
                  
                    toArray() {
                      return this.items.slice().reverse();
                    }
                  }
                  
                  // Esempi di utilizzo con numeri
                  const numStack = new Stack();
                  numStack.push(5);
                  numStack.push(10);
                  numStack.push(15);
                  
                  console.log("Cima dello stack:", numStack.peek()); // Output: 15
                  
                  console.log("Stack prima della pulizia:", numStack.toArray()); // Output: [15, 10, 5]
                  numStack.clear();
                  console.log("Stack dopo la pulizia:", numStack.peek()); // Output: "Lo stack è vuoto."
                  
                  numStack.push(20);
                  numStack.push(25);
                  numStack.push(30);
                  
                  console.log("Posizione di 25 nello stack:", numStack.search(25)); // Output: 2
                  
                  console.log("Stack come array:", numStack.toArray()); // Output: [30, 25, 20]

                class Node {
                    constructor(data) {
                      this.data = data;
                      this.next = null;
                    }
                  }
                  
                  class LinkedList {
                    constructor() {
                      this.head = null;
                    }
                  
                    append(data) {
                      const newNode = new Node(data);
                      if (!this.head) {
                        this.head = newNode;
                      } else {
                        let current = this.head;
                        while (current.next) {
                          current = current.next;
                        }
                        current.next = newNode;
                      }
                    }
                  
                    prepend(data) {
                      const newNode = new Node(data);
                      newNode.next = this.head;
                      this.head = newNode;
                    }
                  
                    delete(data) {
                      if (!this.head) return;
                      if (this.head.data === data) {
                        this.head = this.head.next;
                        return;
                      }
                      let current = this.head;
                      while (current.next) {
                        if (current.next.data === data) {
                          current.next = current.next.next;
                          return;
                        }
                        current = current.next;
                      }
                    }
                  
                    find(data) {
                      let current = this.head;
                      while (current) {
                        if (current.data === data) {
                          return current;
                        }
                        current = current.next;
                      }
                      return null;
                    }
                  
                    display() {
                      let current = this.head;
                      while (current) {
                        console.log(current.data);
                        current = current.next;
                      }
                    }
                  }
                  
                  // Esempio di utilizzo della linked list
                  const myList = new LinkedList();
                  myList.append(1);
                  myList.append(2);
                  myList.append(3);
                  
                  myList.prepend(0);
                  myList.display(); // Output: 0, 1, 2, 3
                  
                  myList.delete(2);
                  myList.display(); // Output: 0, 1, 3
                  
                  const foundNode = myList.find(1);
                  console.log("Nodo trovato:", foundNode ? foundNode.data : "Non trovato"); // Output: Nodo trovato: 1
